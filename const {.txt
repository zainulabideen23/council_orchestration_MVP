const W = 9360;
const C = {
  navy:  '1B3A5C', blue:  '2E75B6', teal:  '0F6E56',

const bdr  = (c = C.bd)   => ({ style: BorderStyle.SINGLE, size: 1, color: c });
const bdrs = (c = C.bd)   => ({ top: bdr(c), bottom: bdr(c), left: bdr(c), right: bdr(c) });
const noBdrs = ()         => ({ top: bdr('FFFFFF'), bottom: bdr('FFFFFF'), left: bdr('FFFFFF'), right: bdr('FFFFFF') });
const CM   = { top: 100, bottom: 100, left: 150, right: 150 };
const CMsm = { top: 60,  bottom: 60,  left: 120, right: 120 };

const run = (t, o = {}) => new TextRun({ text: t, font: 'Arial', size: 22, ...o });
const runB = (t, o = {}) => run(t, { bold: true, ...o });
const runC = (t, o = {}) => new TextRun({ text: t, font: 'Courier New', size: 18, color: C.codetxt, ...o });

const h1 = (t) => new Paragraph({
  heading: HeadingLevel.HEADING_1,
  children: [new TextRun({ text: t, bold: true, size: 34, font: 'Arial', color: C.navy })],
  spacing: { before: 480, after: 180 },
  border: { bottom: { style: BorderStyle.SINGLE, size: 6, color: C.blue, space: 6 } },
});
const h2 = (t) => new Paragraph({
  heading: HeadingLevel.HEADING_2,
  children: [new TextRun({ text: t, bold: true, size: 26, font: 'Arial', color: C.navy })],
  spacing: { before: 320, after: 120 },
});
const h3 = (t) => new Paragraph({
  heading: HeadingLevel.HEADING_3,
  children: [new TextRun({ text: t, bold: true, size: 22, font: 'Arial', color: C.blue })],
  spacing: { before: 220, after: 80 },
});
const p  = (t, o = {}) => new Paragraph({ children: [run(t, o)], spacing: { before: 60, after: 100 } });
const pb = ()           => new Paragraph({ children: [new PageBreak()] });
const sp = ()           => new Paragraph({ children: [], spacing: { before: 120 } });
const note = (t)        => new Paragraph({
  children: [run('Note: ', { bold: true, color: C.amber }), run(t, { color: C.gray })],
  spacing: { before: 80, after: 80 },
  indent: { left: 360 },
  shading: { type: ShadingType.CLEAR, fill: 'FFF8EC' },
  border: { left: { style: BorderStyle.SINGLE, size: 12, color: C.amber } },
});

const bul = (t, lvl = 0) => new Paragraph({
  numbering: { reference: 'bul', level: lvl },
  children: [run(t)],
  spacing: { before: 40, after: 40 },
});
const num = (t, lvl = 0) => new Paragraph({
  numbering: { reference: 'num', level: lvl },
  children: [run(t)],
  spacing: { before: 40, after: 40 },
});
const numB = (t, lvl = 0) => new Paragraph({
  numbering: { reference: 'num', level: lvl },
  children: [runB(t)],
  spacing: { before: 40, after: 40 },
});

const codeBlock = (lines) => lines.map(t => new Paragraph({
  children: [runC(t)],
  spacing: { before: 20, after: 20 },
  indent: { left: 360 },
  shading: { type: ShadingType.CLEAR, fill: C.code },
}));

const mkTable = (headers, rows, widths, bg = C.navy) => new Table({
  width: { size: W, type: WidthType.DXA },
  columnWidths: widths,
  rows: [
    new TableRow({
      tableHeader: true,
      children: headers.map((t, i) => new TableCell({
        borders: bdrs('FFFFFF'),
        width: { size: widths[i], type: WidthType.DXA },
        shading: { fill: bg, type: ShadingType.CLEAR },
        margins: CM,
        verticalAlign: VerticalAlign.CENTER,
        children: [new Paragraph({ children: [new TextRun({ text: t, bold: true, color: 'FFFFFF', font: 'Arial', size: 20 })] })],
      }))
    }),
    ...rows.map((row, ri) => new TableRow({
      children: row.map((t, ci) => new TableCell({
        borders: bdrs(C.bd),
        width: { size: widths[ci], type: WidthType.DXA },
        shading: { fill: ri % 2 === 1 ? C.alt : C.white, type: ShadingType.CLEAR },
        margins: CM,
        children: [new Paragraph({ children: [new TextRun({ text: String(t), font: 'Arial', size: 19, color: '1A2535' })] })],
      }))
    }))
  ]
});

  h1('8. Database Architecture'),
  h2('8.1 Design Decisions'),
  bul('PostgreSQL is used over SQLite because parallel agent writes (N responses per round) require proper row-level locking. SQLite serialises concurrent writes.'),
  bul('UUIDs are used as primary keys throughout. This avoids sequential ID enumeration and simplifies future federation.'),
  bul('All enum types are defined as Prisma enums and enforced at the database level — not just application logic.'),
  bul('The running_briefs table has a UNIQUE constraint on project_id (one brief per project). Updates use upsert, not insert.'),
  bul('stage_council is a junction table with a UNIQUE(stage_id, agent_id) constraint. An agent can be in a stage\'s roster only once.'),
  bul('rounds has UNIQUE(stage_id, round_number) to prevent duplicate round numbers within a stage.'),
  bul('agent_responses has UNIQUE(round_id, agent_id) — one response per agent per round.'),
